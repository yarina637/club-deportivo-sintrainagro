    <script type="module">
        // Importaciones de Firebase
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, setDoc, onSnapshot, collection, query, limit, orderBy } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";
        import { setLogLevel } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // Establecer nivel de log (Opcional, útil para depuración)
        setLogLevel('debug');
        
        // ===========================================
        // 1. INICIALIZACIÓN Y AUTENTICACIÓN DE FIREBASE
        // ===========================================

        const JORNADA_ID = 'jornada_5'; 

        // 1. Obtener la configuración de Firebase
        let firebaseConfig = null;
        // Asignar un token dummy de respaldo si no se proporciona el real.
        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : 'DUMMY_TOKEN_FOR_SANDBOX';
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

        try {
            // Intentar usar la configuración real inyectada por el entorno.
            if (typeof __firebase_config !== 'undefined' && __firebase_config.length > 2) {
                firebaseConfig = JSON.parse(__firebase_config);
            } else {
                // FALLBACK: Configuración dummy para que la aplicación se ejecute en el entorno sin credenciales reales.
                firebaseConfig = {
                    apiKey: "dummy-key",
                    authDomain: "dummy-project.firebaseapp.com",
                    projectId: "dummy-project",
                    storageBucket: "dummy-project.appspot.com",
                    messagingSenderId: "123456789012",
                    appId: "1:123456789012:web:1234567890abcdef"
                };
                console.warn("⚠️ Usando configuración de Firebase DUMMY. La persistencia de datos funcionará, pero no se conectará a una base de datos real sin la configuración inyectada.");
            }
        } catch (e) {
            console.error("Error al parsear la configuración de Firebase. Se usará la configuración DUMMY.", e);
            // Asegurar que la configuración dummy se use incluso si falla el parseo
            firebaseConfig = { apiKey: "dummy-key", projectId: "dummy-project", appId: "1:123456789012:web:1234567890abcdef" };
        }

        let db, auth, userId = null;

        // Inicializar si la configuración (real o dummy) fue exitosa
        const app = initializeApp(firebaseConfig);
        auth = getAuth(app);
        db = getFirestore(app);

        // Intentar autenticación
        onAuthStateChanged(auth, async (user) => {
            if (user) {
                userId = user.uid;
                document.getElementById('user-id-display').textContent = userId;
                console.log('✅ Usuario autenticado:', userId);
                
                // Una vez autenticado, cargar los datos
                loadUserPronosticos();
                loadRanking();
                // Ejecutar siembra de datos (ranking simulado)
                seedRanking(); 

            } else {
                // Si no hay usuario, intentar firmar de forma personalizada (con token real o dummy) o anónima
                try {
                    if (initialAuthToken && initialAuthToken !== 'DUMMY_TOKEN_FOR_SANDBOX') {
                        await signInWithCustomToken(auth, initialAuthToken);
                    } else {
                         // Usar token dummy para sandbox o anónimo si no hay token real
                        await signInAnonymously(auth);
                    }
                } catch (error) {
                    console.error("Error durante la autenticación:", error);
                    displayStatus('error', 'Error de autenticación. No se pudo iniciar sesión en la base de datos.');
                }
            }
        });


        // ===========================================
        // 2. LÓGICA DE CÁLCULO DE PUNTOS
        // ===========================================

        /**
         * Compara los pronósticos con los resultados reales y calcula el puntaje.
         * @param {object} pronosticos - e.g., {score_1_local: '3', score_1_visitante: '1', ...}
         * @returns {number} Puntuación total obtenida.
         */
        function calcularPuntos(pronosticos) {
            let puntosTotales = 0;

            // SIMULACIÓN DE RESULTADOS REALES DE LA JORNADA
            const resultadosReales = {
                'partido_1': { local: 2, visitante: 1 }, 
                'partido_2': { local: 0, visitante: 0 }, 
                'partido_3': { local: 3, visitante: 2 } 
            };

            const partidos = [
                { id: 'partido_1', local: 'score_1_local', visitante: 'score_1_visitante' },
                { id: 'partido_2', local: 'score_2_local', visitante: 'score_2_visitante' },
                { id: 'partido_3', local: 'score_3_local', visitante: 'score_3_visitante' },
            ];

            partidos.forEach(partido => {
                const pronosticoLocal = parseInt(pronosticos[partido.local]);
                const pronosticoVisitante = parseInt(pronosticos[partido.visitante]);
                const realLocal = resultadosReales[partido.id].local;
                const realVisitante = resultadosReales[partido.id].visitante;

                if (isNaN(pronosticoLocal) || isNaN(pronosticoVisitante)) {
                    return;
                }

                // 1. RESULTADO EXACTO (3 PUNTOS)
                if (pronosticoLocal === realLocal && pronosticoVisitante === realVisitante) {
                    puntosTotales += 3;
                    return; 
                }

                // 2. GANADOR/EMPATADOR CORRECTO (1 PUNTO)
                const resultadoReal = Math.sign(realLocal - realVisitante);
                const resultadoPronosticado = Math.sign(pronosticoLocal - pronosticoVisitante);

                if (resultadoPronosticado === resultadoReal) {
                    puntosTotales += 1;
                }
            });

            return puntosTotales;
        }

        // ===========================================
        // 3. FUNCIONES DE FIREBASE (GUARDAR Y CARGAR)
        // ===========================================
        
        /**
         * Guarda los pronósticos del usuario para la jornada actual.
         * @param {object} data - Datos del formulario.
         * @param {number} points - Puntos calculados.
         */
        async function savePronosticos(data, points) {
            if (!userId || !db) {
                console.error("No se puede guardar: Usuario no autenticado o DB no inicializada.");
                return displayStatus('error', 'Error: No se puede guardar. La persistencia de datos está inactiva.');
            }

            // Ruta de datos privados: /artifacts/{appId}/users/{userId}/pronosticos_jornada/{jornadaId}
            const docRef = doc(db, 'artifacts', appId, 'users', userId, 'pronosticos_jornada', JORNADA_ID);

            const pronosticoData = {
                userId: userId,
                jornadaId: JORNADA_ID,
                pronosticos: data,
                puntos_calculados: points,
                timestamp: Date.now()
            };

            try {
                await setDoc(docRef, pronosticoData);
                console.log("✅ Pronósticos guardados en Firestore.");
                displayStatus('success', '¡Tus pronósticos han sido guardados! Puntos calculados (Simulación): ' + points);
                
                // NOTA: En una app real, aquí también actualizarías la tabla 'ranking' (pública)
                // con los puntos totales del usuario.

            } catch (e) {
                console.error("Error al guardar en Firestore: ", e);
                displayStatus('error', 'Error al guardar. Por favor, verifica tus datos e inténtalo de nuevo.');
            }
        }

        /**
         * Carga los pronósticos previos del usuario y llena el formulario.
         */
        function loadUserPronosticos() {
            if (!userId || !db) return;

            const docRef = doc(db, 'artifacts', appId, 'users', userId, 'pronosticos_jornada', JORNADA_ID);
            
            // Usamos onSnapshot para mantener el formulario actualizado 
            onSnapshot(docRef, (docSnap) => {
                if (docSnap.exists()) {
                    const data = docSnap.data().pronosticos;
                    console.log("📝 Pronósticos previos encontrados:", data);
                    
                    // Llenar el formulario con los datos guardados
                    for (const key in data) {
                        const input = document.getElementById(key);
                        if (input) {
                            input.value = data[key];
                        }
                    }
                } else {
                    console.log("🤷‍♂️ No se encontraron pronósticos previos para esta jornada.");
                }
            }, (error) => {
                console.error("Error al cargar pronósticos previos: ", error);
            });
        }
        
        /**
         * Carga el ranking de forma simulada y lo actualiza en tiempo real.
         */
        function loadRanking() {
            if (!db) return;

            // Ruta de datos públicos: /artifacts/{appId}/public/data/ranking
            const rankingRef = collection(db, 'artifacts', appId, 'public', 'data', 'ranking');
            
            // Usar query sin orderBy para evitar errores de índice y ordenar en el cliente
            const q = query(rankingRef, limit(15)); 

            onSnapshot(q, (snapshot) => {
                let rankingData = [];
                snapshot.forEach((doc) => {
                    rankingData.push(doc.data());
                });

                // Ordenar en el cliente (Descendente por puntos_totales)
                rankingData.sort((a, b) => (b.puntos_totales || 0) - (a.puntos_totales || 0));

                renderRanking(rankingData);
            }, (error) => {
                console.error("Error al cargar el ranking: ", error);
                document.getElementById('ranking-body').innerHTML = '<tr><td colspan="3" class="text-center py-4 text-red-500">Error al cargar el ranking.</td></tr>';
            });
        }
        
        /**
         * Renderiza la tabla de ranking.
         * @param {Array<Object>} ranking - Array de objetos de ranking.
         */
        function renderRanking(ranking) {
            const tbody = document.getElementById('ranking-body');
            tbody.innerHTML = ''; // Limpiar

            if (ranking.length === 0) {
                 tbody.innerHTML = '<tr><td colspan="3" class="text-center py-4 text-gray-500">Aún no hay participantes en el ranking.</td></tr>';
                 return;
            }

            ranking.forEach((user, index) => {
                // El userId puede ser el ID anónimo o el ID dummy, pero es único.
                const isCurrentUser = user.userId === userId; 
                const rowClass = isCurrentUser ? 'bg-green-50/50 hover:bg-green-100 border-l-4 border-green-500' : 'hover:bg-gray-50';
                const nameClass = isCurrentUser ? 'font-bold text-green-600' : 'text-gray-900';
                
                const row = `
                    <tr class="${rowClass}">
                        <td class="px-6 py-4 whitespace-nowrap text-sm text-center font-medium">${index + 1}</td>
                        <td class="px-6 py-4 whitespace-nowrap text-sm ${nameClass}">
                            ${isCurrentUser ? '🏆 ' : ''}${user.nombre} ${isCurrentUser ? '(Tú)' : ''}
                        </td>
                        <td class="px-6 py-4 whitespace-nowrap text-sm text-right font-bold text-gray-700">${user.puntos_totales || 0}</td>
                    </tr>
                `;
                tbody.innerHTML += row;
            });
        }

        // ===========================================
        // 4. MANEJO DE EVENTOS DEL FORMULARIO
        // ===========================================

        document.getElementById('formPronosticos').addEventListener('submit', async function (e) {
            e.preventDefault();
            
            if (!userId || !db) {
                return displayStatus('error', 'Error: La aplicación no está lista para guardar. Verifica la conexión a Firebase.');
            }

            const button = document.getElementById('saveButton');
            button.disabled = true;
            button.innerHTML = '<i class="fas fa-spinner fa-spin mr-2"></i> Guardando...';

            // Recoger datos del formulario
            const formData = new FormData(e.target);
            const data = Object.fromEntries(formData.entries());

            // Asegurar que todos los valores son números
            for (const key in data) {
                data[key] = parseInt(data[key]);
            }
            
            // Calcular puntos de simulación
            const puntosObtenidos = calcularPuntos(data);

            // Guardar en Firestore
            await savePronosticos(data, puntosObtenidos);

            // Restablecer botón
            button.disabled = false;
            button.innerHTML = '<i class="fas fa-paper-plane mr-2"></i> Guardar Mis Pronósticos';
        });

        // ===========================================
        // 5. UTILIDADES
        // ===========================================

        /**
         * Muestra el mensaje de estado en la interfaz de usuario.
         * @param {string} type - 'success', 'error', 'warning', 'info'.
         * @param {string} message - El mensaje a mostrar.
         */
        function displayStatus(type, message) {
            const container = document.getElementById('status-message-container');
            const alertClass = {
                success: 'alert-success border-green-400',
                error: 'alert-danger border-red-400',
                warning: 'alert-warning border-yellow-400',
                info: 'alert-info border-blue-400'
            }[type];

            container.innerHTML = `
                <div class="alert ${alertClass} p-4 mb-4 rounded-lg border-l-4 text-center font-bold relative transition-opacity duration-500 opacity-100" role="alert">
                    <i class="fas ${type === 'success' ? 'fa-check-circle' : 'fa-times-circle'} mr-2"></i> ${message}
                    <button type="button" class="absolute top-0 right-0 p-3 text-lg font-semibold leading-none rounded-full" onclick="this.parentElement.remove()">
                        &times;
                    </button>
                </div>
            `;
            
            // Ocultar automáticamente después de 7 segundos
            setTimeout(() => {
                const alertElement = container.querySelector('.alert');
                if (alertElement) {
                    alertElement.classList.remove('opacity-100');
                    alertElement.classList.add('opacity-0');
                    setTimeout(() => alertElement.remove(), 500);
                }
            }, 7000);
        }

        // ===========================================
        // 6. CÓDIGO DE INICIALIZACIÓN DE DATOS DE RANKING (Simulado)
        // Solo para demostrar que el ranking funciona con onSnapshot
        // ===========================================

        async function seedRanking() {
            if (!db || !userId) return; // Asegurar que la DB y el ID de usuario estén listos

            const seedData = [
                { userId: 'sim_1', nombre: 'Laura C. (Finca El Roble)', puntos_totales: 138 },
                { userId: 'sim_2', nombre: 'Javier M. (Seguridad)', puntos_totales: 130 },
                { userId: 'sim_3', nombre: 'Ana P. (Logística)', puntos_totales: 112 },
                { userId: 'sim_4', nombre: 'Carlos R. (Mecánica)', puntos_totales: 105 },
                { userId: 'sim_5', nombre: 'María Z. (Comercial)', puntos_totales: 98 },
            ];

            const rankingCollectionRef = collection(db, 'artifacts', appId, 'public', 'data', 'ranking');

            for (const data of seedData) {
                const docRef = doc(rankingCollectionRef, data.userId);
                try {
                    await setDoc(docRef, data, { merge: true }); 
                } catch (e) {
                    console.error("Error al sembrar datos de ranking:", e);
                }
            }
            
            // Añadir/Actualizar el usuario actual con puntos base
            const userDocRef = doc(rankingCollectionRef, userId);
            await setDoc(userDocRef, { 
                userId: userId, 
                // Usar el ID anónimo/dummy del usuario para el nombre simulado
                nombre: `Jugador ${userId.substring(0, 5)} (Tú)`, 
                puntos_totales: 145 
            }, { merge: true });
        }
    </script>
