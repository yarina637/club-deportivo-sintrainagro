<div class="container py-5">
    <div id="trivia-container">
        <h1 class="display-5 fw-bold text-info mb-4">🏆 Trivia de Deportes</h1>
        <p class="lead">Responde las siguientes preguntas para demostrar tus conocimientos deportivos.</p>
        
        <div id="quiz-area">
            </div>

        <p id="score-display" class="fs-4 fw-bold mt-4"></p>
        
        <button id="submit-btn" class="btn btn-primary btn-lg mt-3">Terminar y Ver Puntuación</button>
        <button id="restart-btn" class="btn btn-danger btn-lg mt-3 d-none" onclick="startTrivia()">Reiniciar Trivia</button>
    </div>
</div>

<script>
    // 1. Definición del array de preguntas
    const questions = [
        {
            question: "¿Cuántos jugadores tiene un equipo de fútbol en la cancha durante un partido?",
            answers: ["10", "11", "12", "9"],
            correctAnswer: "11"
        },
        {
            question: "¿Qué deporte utiliza la palabra 'birdie'?",
            answers: ["Tenis", "Bádminton", "Golf", "Voleibol"],
            correctAnswer: "Golf"
        },
        {
            question: "¿Cuál es la distancia de un maratón en kilómetros?",
            answers: ["35.195 km", "40.2 km", "42.195 km", "50 km"],
            correctAnswer: "42.195 km"
        },
        {
            question: "¿En qué país se originó el Taekwondo?",
            answers: ["China", "Japón", "Corea del Sur", "Tailandia"],
            correctAnswer: "Corea del Sur"
        }
    ];

    let currentScore = 0;

    const quizArea = document.getElementById('quiz-area');
    const scoreDisplay = document.getElementById('score-display');
    const submitBtn = document.getElementById('submit-btn');
    const restartBtn = document.getElementById('restart-btn');

    // Función para iniciar o reiniciar el juego
    function startTrivia() {
        currentScore = 0;
        quizArea.innerHTML = ''; // Limpia el área de preguntas
        scoreDisplay.textContent = 'Puntuación: 0';
        submitBtn.classList.remove('d-none');
        restartBtn.classList.add('d-none');
        submitBtn.disabled = false;
        
        // Renderiza cada pregunta
        questions.forEach((q, index) => {
            const card = document.createElement('div');
            card.classList.add('question-card');
            card.innerHTML = `
                <p class="fw-bold">Q${index + 1}: ${q.question}</p>
                ${q.answers.map(answer => `
                    <button class="btn btn-outline-info btn-answer" 
                            data-q-index="${index}" 
                            data-answer="${answer}">
                        ${answer}
                    </button>
                `).join('')}
            `;
            quizArea.appendChild(card);
        });

        // Añade el listener para manejar las selecciones de respuesta
        document.querySelectorAll('.btn-answer').forEach(button => {
            button.addEventListener('click', handleAnswerClick);
        });
    }

    // Maneja la selección de una respuesta
    function handleAnswerClick(e) {
        const selectedBtn = e.target;
        const qIndex = selectedBtn.dataset.qIndex;

        // Deshabilita todos los botones de la pregunta actual para evitar doble click
        document.querySelectorAll(`.btn-answer[data-q-index="${qIndex}"]`).forEach(btn => {
            btn.disabled = true;
            btn.classList.remove('btn-outline-info');
        });

        // Resalta la respuesta seleccionada
        if (selectedBtn.dataset.answer === questions[qIndex].correctAnswer) {
            selectedBtn.classList.add('correct');
        } else {
            selectedBtn.classList.add('incorrect');
            // Si es incorrecto, también resaltamos la respuesta correcta (opcional)
            document.querySelector(`.btn-answer[data-q-index="${qIndex}"][data-answer="${questions[qIndex].correctAnswer}"]`).classList.add('correct');
        }
    }

    // Evalúa el resultado final
    submitBtn.addEventListener('click', () => {
        currentScore = 0;
        let answeredCount = 0;

        questions.forEach((q, index) => {
            const correctBtn = document.querySelector(`.btn-answer[data-q-index="${index}"].correct`);
            
            // Si se encontró la respuesta correcta marcada con la clase 'correct'
            if (correctBtn) {
                // Si la respuesta seleccionada coincide con la respuesta correcta Y está marcada como 'correct'
                // Esto es una simplificación, ya que el handleAnswerClick ya marca la correcta.
                
                // Buscamos la respuesta del usuario (la que es correcta o incorrecta)
                const userSelectedBtn = document.querySelector(`.btn-answer[data-q-index="${index}"].correct:not(.btn-outline-info), .btn-answer[data-q-index="${index}"].incorrect`);
                
                // Si el usuario seleccionó la respuesta correcta, sumamos punto
                if (correctBtn.dataset.answer === q.correctAnswer) {
                    currentScore++;
                }
                
                answeredCount++;
            }
        });
        
        // Deshabilita el botón de enviar y muestra el botón de reiniciar
        submitBtn.disabled = true;
        restartBtn.classList.remove('d-none');
        
        // Muestra el resultado
        scoreDisplay.textContent = `Puntuación Final: ${currentScore} de ${questions.length}. ¡Muy bien!`;

        // Deshabilita todos los botones para evitar más cambios
        document.querySelectorAll('.btn-answer').forEach(btn => btn.disabled = true);
    });

    // Iniciar el juego al cargar la página
    startTrivia();
</script>
