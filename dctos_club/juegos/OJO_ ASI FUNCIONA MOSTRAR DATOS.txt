<script>
    // -------------------------------------------------------------
    // FUNCIÓN: Filtra Pagos (recarga la página con los parámetros de la URL)
    // -------------------------------------------------------------
    function filtrarPagosTotal() {
        const finca = document.getElementById('finca-select').value;
        const catorcena = document.getElementById('catorcena-select').value;
        
        let url = '/pagos?';
        
        if (finca) {
            url += `finca=${finca}&`;
        }
        if (catorcena) {
            url += `catorcena=${catorcena}&`;
        }
        
        window.location.href = url.endsWith('&') ? url.slice(0, -1) : url;
    }

    // -------------------------------------------------------------
    // FUNCIÓN: Carga los datos de la Finca en los inputs y filtra
    // -------------------------------------------------------------
    function cargarDatosFinca() {
        const fincaSeleccionada = document.getElementById('finca-select').value;
        
        // 1. Si se selecciona "Todas", limpiamos los campos y solo filtramos la tabla.
        if (!fincaSeleccionada || fincaSeleccionada === '') {
            limpiarCamposFinca();
            filtrarPagosTotal(); 
            return;
        }

        const url = `/pagos/info-finca?finca=${fincaSeleccionada}`;

        // 2. Llamada asíncrona a la API de Node.js
        fetch(url)
            .then(response => response.json())
            .then(data => {
                if (data && data.finca) {
                    // 3. Rellenar los inputs con los IDs únicos
                    document.getElementById('razon').value = data.razon_social || ''; 
                    document.getElementById('nit').value = data.nit || '';
                    document.getElementById('repleg').value = data.representante_legal || '';
                    document.getElementById('dir').value = data.direccion || '';
                    document.getElementById('tel').value = data.telefono || '';
                    document.getElementById('correo').value = data.correo || '';
                    document.getElementById('mpio').value = data.municipio || '';
                    document.getElementById('dpto').value = data.departamento || ''; 
                    document.getElementById('input-seccional').value = data.seccional || ''; // Usa el ID corregido
                    document.getElementById('estfca').value = data.estado_finca || '';
                    document.getElementById('obsfca').value = data.observaciones || '';
                } else {
                    limpiarCamposFinca();
                    console.warn(`No se encontró información para la finca: ${fincaSeleccionada}`);
                }
                
                // Llama al filtro principal después de cargar los datos
                filtrarPagosTotal(); 
            })
            .catch(error => {
                console.error('Error al obtener datos de la finca:', error);
                limpiarCamposFinca();
                filtrarPagosTotal(); 
            });
    }

    // Función auxiliar para limpiar los campos
    function limpiarCamposFinca() {
        document.getElementById('razon').value = '';
        document.getElementById('nit').value = '';
        document.getElementById('repleg').value = '';
        document.getElementById('dir').value = '';
        document.getElementById('tel').value = '';
        document.getElementById('correo').value = '';
        document.getElementById('mpio').value = '';
        document.getElementById('dpto').value = '';
        document.getElementById('input-seccional').value = ''; // Usa el ID corregido
        document.getElementById('estfca').value = '';
        document.getElementById('obsfca').value = '';
    }
</script>
