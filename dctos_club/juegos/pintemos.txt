<script>
    const coloringSvg = document.getElementById('coloring-svg');
    const colorableParts = coloringSvg.querySelectorAll('.colorable'); // Todas las partes que se pueden pintar
    const colorSwatches = document.querySelectorAll('.color-swatch');
    const resetButton = document.getElementById('reset-button');
    const messageElement = document.getElementById('message');

    let selectedColor = '#e74c3c'; // Color inicial por defecto (rojo)

    // Función para aplicar el color seleccionado a una parte del SVG
    function paintPart(event) {
        // Solo si se hizo clic en una parte que tenga la clase 'colorable'
        if (event.target.classList.contains('colorable')) {
            event.target.style.fill = selectedColor;
            messageElement.textContent = `¡Pintaste el ${event.target.id.replace('svg-', '')} de ${selectedColor}!`;
        }
    }

    // Función para seleccionar un color de la paleta
    function selectColor(event) {
        // Quitar la clase 'selected' de todos los colores
        colorSwatches.forEach(swatch => swatch.classList.remove('selected'));
        // Añadir la clase 'selected' al color clicado
        event.target.classList.add('selected');
        selectedColor = event.target.dataset.color;
        messageElement.textContent = `Color seleccionado: ${selectedColor}. ¡Ahora haz clic en el dibujo!`;
    }

    // Función para reiniciar el dibujo a su estado original
    function resetDrawing() {
        colorableParts.forEach(part => {
            part.style.fill = '#ecf0f1'; // Restaurar el color gris inicial
        });
        messageElement.textContent = '¡Dibujo reiniciado! Selecciona un color y vuelve a pintar.';
        // Asegurarse de que el color rojo siga siendo el seleccionado por defecto o el último
        colorSwatches.forEach(swatch => swatch.classList.remove('selected'));
        document.querySelector('.color-swatch[data-color="#e74c3c"]').classList.add('selected');
        selectedColor = '#e74c3c';
    }

    // --- Configuración de Eventos Iniciales ---

    // 1. Añadir listeners para cada parte pintable del SVG
    colorableParts.forEach(part => {
        part.addEventListener('click', paintPart);
    });

    // 2. Añadir listeners para cada botón de color
    colorSwatches.forEach(swatch => {
        swatch.addEventListener('click', selectColor);
    });

    // 3. Añadir listener para el botón de Reiniciar
    resetButton.addEventListener('click', resetDrawing);

    // 4. Seleccionar el primer color por defecto al cargar la página
    document.querySelector('.color-swatch[data-color="#e74c3c"]').classList.add('selected');
    messageElement.textContent = `Color inicial: Rojo. ¡Haz clic en el dibujo para empezar!`;
</script>
