<script>
    // **********************************************
    // LÓGICA DEL JUEGO DE SECUENCIA
    // **********************************************

    // --- Variables Globales ---
    const cells = document.querySelectorAll('.cell');
    const startButton = document.getElementById('start-button');
    const levelElement = document.getElementById('level');
    const messageElement = document.getElementById('message');

    let sequencePC = [];
    let sequencePlayer = [];
    let level = 1;
    let canClick = false; 
    let isPlaying = false; 

    // --- Funciones del Juego ---

    function nextLevel() {
        canClick = false;
        sequencePlayer = [];
        levelElement.textContent = level;
        messageElement.textContent = '¡Observa con atención!';
        
        // Genera el nuevo paso aleatorio (0, 1, 2 o 3)
        const randomCellId = Math.floor(Math.random() * cells.length);
        sequencePC.push(randomCellId);
        
        // Espera y reproduce la secuencia
        setTimeout(playSequence, 1000); 
    }

    function playSequence() {
        let i = 0;
        const interval = setInterval(() => {
            const idToActivate = sequencePC[i];
            
            const cell = document.querySelector(`[data-id="${idToActivate}"]`);
            activateCell(cell);
            
            i++;
            if (i >= sequencePC.length) {
                clearInterval(interval);
                
                // Habilita el click del jugador
                canClick = true;
                messageElement.textContent = '¡Tu turno! Repite la secuencia.';
            }
        }, 700); // Velocidad de la secuencia (700ms por paso)
    }

    function activateCell(cell) {
        cell.classList.add('active-flash');
        // Desactiva el brillo después de un flash rápido
        setTimeout(() => {
            cell.classList.remove('active-flash');
        }, 350); 
    }

    function checkPlayerInput(clickedCellId) {
        sequencePlayer.push(parseInt(clickedCellId));

        const index = sequencePlayer.length - 1;

        // 1. Comprueba si el último clic fue correcto
        if (sequencePlayer[index] !== sequencePC[index]) {
            gameOver();
            return;
        }

        // 2. Comprueba si la secuencia está completa
        if (sequencePlayer.length === sequencePC.length) {
            level++;
            messageElement.textContent = '✅ ¡Correcto! Preparando Nivel ' + level;
            setTimeout(nextLevel, 1500); // Espera 1.5s antes del siguiente nivel
        }
    }

    function gameOver() {
        isPlaying = false;
        canClick = false;
        messageElement.textContent = `❌ ¡Fin del juego! Llegaste al Nivel ${level}.`;
        startButton.textContent = 'Volver a Jugar';
        startButton.disabled = false;
        
        // Resetear el estado
        sequencePC = [];
        level = 1;
    }

    function handleCellClick(event) {
        // Ignora el click si no es el turno del jugador o si no es una celda
        if (!canClick || !event.target.classList.contains('cell')) return; 
        
        const clickedCellId = event.target.dataset.id;
        
        activateCell(event.target);
        
        checkPlayerInput(clickedCellId);
    }

    function startGame() {
        if (isPlaying) return; 

        isPlaying = true;
        startButton.disabled = true; // Deshabilita el botón mientras el juego está activo
        startButton.textContent = 'En curso...';
        
        nextLevel();
    }

    // --- Configuración de Eventos Iniciales ---

    // Añade el listener de clic para todas las celdas
    document.getElementById('game-board').addEventListener('click', handleCellClick);
    
    // Listener para el botón de inicio
    startButton.addEventListener('click', startGame);

</script>
