<div class="container py-4">
    <div id="game-container">
        <h1>🧠 Mini-Cerebros: Memorama de Emojis</h1>
        <div id="game-board">
        </div>
        <p id="message">¡Encuentra todas las parejas!</p>
        <button onclick="startGame()" class="btn btn-danger fw-bold">Reiniciar Juego</button>
    </div>
</div>

<script>
    // **********************************************
    // ** A S E G Ú R A T E   D E   C O P I A R   **
    // ** T O D A   L A   L Ó G I C A   J S   A Q U Í **
    // **********************************************
    
    // 1. Array de Emojis para el juego
    const emojis = [
        '😀', '😎', '🤩', '🥳', '🤖', '👾', '🚀', '💡',
        '😀', '😎', '🤩', '🥳', '🤖', '👾', '🚀', '💡'
    ]; 

    let gameBoard = document.getElementById('game-board');
    let messageDisplay = document.getElementById('message');
    let flippedCards = []; 
    let matchedPairs = 0;

    // Función para barajar el array
    function shuffleArray(array) {
        for (let i = array.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [array[i], array[j]] = [array[j], array[i]];
        }
    }

    // 2. Función principal para inicializar el juego
    function startGame() {
        matchedPairs = 0;
        flippedCards = [];
        gameBoard.innerHTML = ''; 
        messageDisplay.textContent = '¡Encuentra todas las parejas!';
        
        shuffleArray(emojis);

        shuffleArray(emojis);

        emojis.forEach((emoji, index) => {
            const card = document.createElement('div');
            card.classList.add('card');
            card.dataset.emoji = emoji;
            card.dataset.index = index;
            card.textContent = emoji; 
            
            // 🚨 AÑADE ESTA LÍNEA PARA FORZAR EL OCULTAMIENTO DEL ICONO 🚨
            card.style.color = 'transparent'; 
            
            card.addEventListener('click', () => flipCard(card));
            gameBoard.appendChild(card);
        });
    }

    // 3. Función para voltear una carta
    function flipCard(card) {
        if (flippedCards.length < 2 && !card.classList.contains('flipped')) {
            card.classList.add('flipped');
            card.style.color = '#333'; 
            flippedCards.push(card);

            if (flippedCards.length === 2) {
                document.querySelectorAll('.card').forEach(c => c.style.pointerEvents = 'none');
                
                setTimeout(checkForMatch, 1000); 
            }
        }
    }

    // 4. Función para verificar si hay coincidencia
    function checkForMatch() {
        const [card1, card2] = flippedCards;

        if (card1.dataset.emoji === card2.dataset.emoji) {
            // ¡Coincidencia!
            card1.classList.add('matched');
            card2.classList.add('matched');
            card1.style.color = 'white';
            card2.style.color = 'white';
            matchedPairs++;
            messageDisplay.textContent = `¡Pareja encontrada! Llevas ${matchedPairs} de 8.`;

            if (matchedPairs === emojis.length / 2) {
                messageDisplay.textContent = '🎉 ¡Felicitaciones! Has encontrado todas las parejas. 🎉';
            }
        } else {
            // No hay coincidencia
            card1.classList.remove('flipped');
            card2.classList.remove('flipped');
            card1.style.color = 'transparent';
            card2.style.color = 'transparent';
            messageDisplay.textContent = '¡Inténtalo de nuevo!';
        }

        // Limpiar y re-habilitar
        flippedCards = [];
        document.querySelectorAll('.card:not(.matched)').forEach(c => c.style.pointerEvents = 'auto');
    }

    // Iniciar el juego
    startGame();
</script>
